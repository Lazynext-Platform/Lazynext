# Build instructions:
# 1. Install deps: apt-get install -y build-essential pkg-config libssl-dev python3 python3-pip unzip ffmpeg
# 2. Install Bun: curl -fsSL https://bun.sh/install | bash
# 3. Install Rust: curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh -s -- -y
# 4. Build Rust services: cargo build --release -p lazynext_api_gateway -p collab-server
# 5. Build web: cd apps/web && bun run build (standalone output)
# 6. Python venvs: cd services/pre-processing && python3 -m venv venv && source venv/bin/activate && pip install -r requirements.txt
# 7. Copy systemd files, run: systemctl daemon-reload && systemctl enable --now lazynext-*
